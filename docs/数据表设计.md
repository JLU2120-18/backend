# Prophet Salary 数据表设计

## 用户
```typescript
interface Auth {
  id: string;
  username: string;
  password: string;
  role: 'employee' | 'commission' | 'payroll';
}

// JWT 携带的信息，作为 JWT 携带的信息而不用作为数据表
interface JWT {
  id: string;
  role: string;
}

interface User {
  id: string;
  username: string;
  address: string;
  socsec_id: string;
  tax_rate: number;
  other_cast: number;
  phone: string;
  hour_wage?: number;
  salary?: decimal;
  commission_rate?: number;
  duration_limit: number; // 0 代表可以加班
  type?: 'salary' | 'commission' | 'wage';
  payment: 'mail' | 'receive' | 'bank';
  mail_address?: string;
  bank_name?: string;
  bank_account?: string;
}
```

## 各种报告(作为类型，不作为数据库表)
```typescript
interface EmployeeDurationReport {
  employee_id: string;
  employee_name: string;
  start_time: string;
  end_time: string;
  duration: number;
}

interface EmployeeSalaryReport {
  employee_id: string;
  employee_name: string;
  start_time: string;
  end_time: string;
  salary: decimal;
}

interface DurationReport {
  employee_id: string;
  employee_name: string;
  start_time: string;
  end_time: string;
  duration: number;
}

interface ProjectDurationReport {
  employee_id: string;
  employee_name: string;
  start_time: string;	
  end_time: string;
  data: {
    project_name: string;
    duration: number;
  }[];
}

interface VacationReport {
  employee_id: string;
  employee_name: string;
  start_time: string;
  end_time: string;
  days: number;
}

interface SalaryReport {
  employee_id: string;
  employee_name: string;
  start_time: string;
  end_time: string;
  salary: decimal;
}
```
